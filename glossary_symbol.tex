\newcommandx\acr[5][4=,5=]{
  \ifthenelse{\equal{#5}{}}
  {
    \acrSing{#1}{#2}{#3}
  }
  {
    \acrPl{#1}{#2}{#3}{#4}{#5}
  }
}

\newcommandx\symbl[4][4=,]{
  \ifthenelse{\equal{#4}{}}
  {
    \symblnosort{#1}{#2}{#3}
  }
  {
    \symblwithsort{#1}{#2}{#3}{#4}
  }
}

}
